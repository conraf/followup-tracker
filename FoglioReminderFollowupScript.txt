Vuoi provare a impostare un orario specifico o ti trovi bene con gli eventi "tutto il giorno"?

Per impostare un orario specifico (ad esempio le 09:00 del mattino), dobbiamo cambiare il comando da createAllDayEvent a createEvent e definire con precisione l'ora di inizio e di fine.
Ecco il codice finale completo e pronto all'uso. Sostituisci tutto quello che hai nell'editor con questo:

================

function creaReminderCalendar() {
  var foglio = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var dati = foglio.getDataRange().getValues();
  
  // INSERISCI QUI LA TUA EMAIL
  var idCalendario = 'tua-email@gmail.com'; 
  var calendario = CalendarApp.getCalendarById(idCalendario);
  
  for (var i = 1; i < dati.length; i++) {
    var nome = dati[i];           // Colonna A
    var valoreCellaD = dati[i];    // Colonna D
    var azione = dati[i];         // Colonna C
    
    var dataInizio = new Date(valoreCellaD);
    
    if (nome && !isNaN(dataInizio.getTime())) {
      
      // IMPOSTA L'ORARIO: qui impostiamo le 09:00 del mattino
      dataInizio.setHours(9);
      dataInizio.setMinutes(0);
      dataInizio.setSeconds(0);
      
      // L'evento durerà 30 minuti (fino alle 09:30)
      var dataFine = new Date(dataInizio);
      dataFine.setMinutes(30);

      var titolo = "⚠️ Follow-up: " + nome;
      var descrizione = "Azione richiesta: " + azione + "\nCreato automaticamente da Fogli.";

      // Controlliamo se esiste già un evento con lo stesso titolo in quel giorno
      var eventi = calendario.getEvents(dataInizio, dataFine, {search: titolo});
      
      if (eventi.length === 0) {
        calendario.createEvent(titolo, dataInizio, dataFine, {
          description: descrizione
        });
        Logger.log("✅ Creato evento alle 09:00 per: " + nome);
      } else {
        Logger.log("ℹ️ Riga " + (i+1) + " già presente alle 09:00.");
      }
    }
  }
}

================

Visibilità: Vedrai il reminder proprio come un appuntamento nel tuo calendario giornaliero alle ore 9:00.
Personalizzazione orario: Se preferisci un altro orario, ti basta cambiare i numeri in setHours(9) e setMinutes(0).
